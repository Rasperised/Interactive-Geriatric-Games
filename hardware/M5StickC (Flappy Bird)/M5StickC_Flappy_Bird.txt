#include <M5StickC.h>

void setup() {
  M5.begin();
  Serial.begin(115200);
  M5.Imu.Init();
}

void loop() {
  float accX, accY, accZ;
  M5.Imu.getAccelData(&accX, &accY, &accZ);

  // We'll use pitch (forward/backward tilt)
  // Typically tilting forward changes Y axis
  Serial.print(accY);  
  Serial.print(",");
  Serial.println(accZ); 

  delay(100); // send 10 times per second
}
